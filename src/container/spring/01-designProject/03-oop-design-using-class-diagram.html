oop design using class diagram

        Artifact                                      Wizard
     --------------                               --------------
     id                                           id
     name            0 ~ n  _________  0 ~ 1      name
     description     ownedArtifact __ owner
     imageUrl
                     wizard는 0~n개의 artifact를 갖고
                     artifact는 0~1개의 wizard를 갖음

class Artifact

    @Getter
    @Setter
    @NoArgumentConstructor
    @Entity
    public class Artifact implements Serializable {
      @Id
      private String id;
      private String name;
      private String description;
      private String imageUrl;
      @ManyToOne
      private Wizard owner;
    }

class Wizard

    @Getter
    @Setter
    @NoArgumentConstructor
    @Entity
    public class Wizard implements Serializable {
      @Id
      private Integer id;
      private String name;
      @OneToMany(casecase = {CasecadeType.PERSIST, CasecadeType.MERGE}, mappedBy = "owner", )
      private List[Artifact] artifacts;
    }
